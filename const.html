function doGet() { try { // Obtener datos de la hoja de cálculo const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('Inventario Drive'); if (!sheet) { return createErrorResponse('No se encontró la hoja "Inventario Drive"'); } // Obtener todos los datos de la hoja const data = sheet.getDataRange().getValues(); // Si no hay datos (solo encabezados o vacío) if (data.length <= 1) { return createErrorResponse('No hay datos en el inventario. Ejecuta primero listarArchivosDrive()'); } // Extraer encabezados y datos const headers = data[0]; const rows = data.slice(1); // Crear estructura de datos para la web const filesData = rows.map((row, index) => { const file = { id: index + 1, nombre: row[1] || '', carpeta: row[2] || '', clasificacion: row[3] || 'Gestión Documental', fecha: row[4] || '', miniatura: row[5] || '', titulo: row[6] || '', duracion: row[7] || '' }; return file; }); // Crear página HTML const htmlOutput = HtmlService.createTemplateFromFile('Index'); htmlOutput.files = filesData; htmlOutput.headers = headers; return htmlOutput.evaluate() .setTitle('Inventario de Documentos - Drive') .addMetaTag('viewport', 'width=device-width, initial-scale=1') .setFaviconUrl('https://drive.google.com/favicon.ico'); } catch (error) { console.error('Error en doGet:', error); return createErrorResponse(`Error al cargar los datos: ${error.message}`); } } // Función para crear respuesta de error function createErrorResponse(message) { const html = `
